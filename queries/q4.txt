#Query 4 (list all courses offered by the university filtered by subject)
#example: <https://www.concordia.ca/>
SELECT ?course
  WHERE{
	 ?course <http://example.org/course/offeredBy> [university_name].
  	?course <http://example.org/course/courseSubject> ?subject
  FILTER (?subject = "COMP")
  }
	LIMIT 100
