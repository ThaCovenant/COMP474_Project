#Query 10 (list all topics offered by the course)
#example: <http://example.org/course/COMP474>
SELECT DISTINCT ?topic
  WHERE{
 <http://example.org/course/{topic}> <http://example.org/course/hasLecture> ?lecture.
  ?lecture <http://example.org/lecture/hasContent> ?material.
 ?topic <http://example.org/topic/topicProvenanceInformation> ?material
  }

 #[course] <http://example.org/course/hasLecture> ?lecture.
