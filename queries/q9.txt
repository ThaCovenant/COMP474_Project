#Query 9 (list all materials of a course filtered by topic)
#example: <http://example.org/course/COMP474>
#example: <http://example.org/topic/KnowledgeGraph>
SELECT ?content
  WHERE{
<http://example.org/course/{course}> <http://example.org/course/hasLecture> ?lectures.
?lectures <http://example.org/lecture/hasContent> ?content.
?topic <http://example.org/topic/topicProvenanceInformation> ?content.
  FILTER(?topic = <http://example.org/topic/{topic}>)
  }
	LIMIT 100
