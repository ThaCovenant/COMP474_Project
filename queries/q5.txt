#Query 5 (list all materials of a course filtered by topic)(**doesnt filter by topic)
#example: <http://example.org/course/COMP335>
SELECT ?subject
  WHERE{
[course] <http://example.org/course/hasLecture> ?lectures.
?lectures <http://example.org/lecture/hasContent> ?content.
?subject <http://www.w3.org/2000/01/rdf-schema#subClassOf> ?content.
  }
	LIMIT 100
