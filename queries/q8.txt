#Query 8 (list all material covered in the course filtered by lecture number)
#example: <http://example.org/course/COMP474> <http://example.org/lecture/COMP474Lecture3>
SELECT ?content
  WHERE{
<http://example.org/course/{course}> <http://example.org/course/hasLecture> ?lectures.
?lectures <http://example.org/lecture/hasContent> ?content.
FILTER(?lectures = <http://example.org/lecture/{lecture}>).
  }
	LIMIT 100
