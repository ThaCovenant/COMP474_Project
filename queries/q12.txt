#Query 12 (list all students that have completed a course)
#example: <http://example.org/course/COMP474>
SELECT DISTINCT ?student_id ?firstName ?lastName
WHERE {
?id <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://example.org/course/{course}>.
?student_id <http://example.org/student/hasCompletedCourse> ?id.
?student_id <http://example.org/student/hasFirstName> ?firstName.
?student_id <http://example.org/student/hasLastName> ?lastName
}
